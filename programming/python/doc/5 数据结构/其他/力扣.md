# [190. 颠倒二进制位](https://leetcode.cn/problems/reverse-bits/description/)
# 方案一 转换成列表
```python
class Solution:
    def reverseBits(self, n: int) -> int:
        print(f"n: {n}")
        li = list(str(bin(n)[2:].zfill(32)))
        # li = reversed(li)
        print(f"li: {li}")
        li.reverse()
        print(f"li: {li}")
        print(f"len(li): {len(li)}")
        sum = 0
        for i in range(len(li)):
            sum += int(li[i]) * (2**(len(li)-1-i))
        return sum
```

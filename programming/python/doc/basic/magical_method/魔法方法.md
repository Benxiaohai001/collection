# __init__()
类型初始化时调用
# __call__()
当一个类的实例像函数一样被调用时，__call__方法会自动执行
```python
# __call__方法
class MyCallable:
    def __init__(self, value):
        self.value = value
    def __call__(self, increment):
        self.value += increment
        return self.value

# 创建类的实例
callable = MyCallable(10)

# 像函数一样调用实例
result = callable(5)
print(f"1st {result}")  # 15

# 再次调用, 在上次调用的基础上再次调用
result = callable(10)
print(f"2rd {result}")  # 25
```
